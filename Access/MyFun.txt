Второй пример применения метода Delete

Фрагмент программы данного примера добавляет запись в таблицу "Клиенты" текущей базы данных "Борей" (файл NWIND.MDB), а затем удаляет ее с помощью метода Delete.

Dim МояБД As Database, МойНабор As Recordset
Set МояБД = DBEngine.Workspaces(0).Databases(0)   ' Определение текущей базы данных.
Set МойНабор = МояБД![Клиенты].OpenRecordset()   ' Открытие таблицы.
МойНабор.Index = "PrimaryKey"	' Определить индекс.

МойНабор.AddNew		' Подготовка новой записи.
МойНабор![Клиент] = "ЗМЕЙК"	' Задание ключа.
МойНабор![Название] = "Кафе ""Зеленый змей"" "   ' Задание названия фирмы.
МойНабор.Update		' Сохранение изменений.
...
МойНабор.Seek "=", "ЗМЕЙК"	' Определение текущей записи.
If Not МойНабор.NoMatch Then
	МойНабор.Delete	' Удаление записи.
End If

МойНабор.Close		' Закрытие таблицы.
